SCORING ALGORITHM
python# TOXICITY SCORE (0-100)
base_score = sum(Q1-Q10 weighted responses)  # Max 80
+ frequency_score (Q11)  # Max 10
+ violence (Q14)  # Max 15
+ control (Q19-Q22)  # Max 10 each
+ legal threats (Q24-Q26)  # Max 15

Total possible: 80+10+15+40+15 = 160 (scale to 100)

# NARCISSISM TYPE
if (Q4="Usually/Always" AND Q7="Often/Constantly" AND Q9="Often/Pattern"):
    type = "Covert Narcissist"
elif (Q14="Physical" AND Q6="Operates" AND Q22="No privacy"):
    type = "Overt Narcissist"
elif (Q13="Rapidly" AND Q18="Believe promises" AND unstable pattern):
    type = "Borderline Traits"
elif (score < 40):
    type = "Toxic but not NPD"

# DANGER LEVEL
if (Q14="Injury/weapon" OR Q24="Police report" OR Q26="Multiple/RO"):
    danger = "SEVERE"
elif (Q14="Physical" OR Q24="Threatened police" OR Q23="Threatens custody"):
    danger = "HIGH"
elif (Q13="Significantly worse" AND score > 60):
    danger = "MEDIUM"
else:
    danger = "LOW"

# DETECTED TACTICS (for AI narrative)
tactics = []
if Q1 >= "Often": tactics.append("gaslighting")
if Q3 >= "Often": tactics.append("silent_treatment")
if Q4 >= "Usually": tactics.append("victim_playing")
if Q8 >= "Often": tactics.append("triangulation")
if Q6 >= "Often": tactics.append("manipulation")

# VULNERABILITIES (for AI explanation)
vulnerabilities = []
if Q15 >= "Often": vulnerabilities.append("high_empathy")
if Q16 == "Avoid/peace": vulnerabilities.append("conflict_avoidant")
if Q17 >= "Often": vulnerabilities.append("high_self_doubt")
if Q18 >= "Yes, with": vulnerabilities.append("change_belief")

# EXIT BARRIERS
barriers = []
if Q19 >= "Manages": barriers.append("financial_control")
if Q20 >= "Significantly": barriers.append("isolation")
if Q23 >= "Yes, child": barriers.append("children")
if Q27 >= "Shared mortgage": barriers.append("property")
if Q30 <= "Maybe 1-2": barriers.append("no_support")

# RISK FLAGS
risks = []
if Q23 >= "Uses against": risks.append("custody_weaponization")
if Q24 >= "Threatened police": risks.append("false_accusation_risk")
if Q25 >= "Work contacts": risks.append("reputation_attack")
if Q14 != "Never": risks.append("violence_history")

# EXIT TIMELINE
if Q27 >= "Shared property" AND Q28 >= "Shared mortgage":
    timeline = "12-month"
elif Q27 >= "Shared lease" OR Q23 == "Yes, child":
    timeline = "6-month"
elif Q27 == "Living together":
    timeline = "3-month"
else:
    timeline = "immediate_option"

# RECOMMENDED ACTION
if danger == "SEVERE":
    action = "emergency_exit"
elif danger == "HIGH":
    action = "exit_planning"
elif score >= 60:
    action = "document_and_prepare"
elif score >= 40:
    action = "boundaries_and_monitor"
else:
    action = "observe_and_assess"
